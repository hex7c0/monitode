monitode

resource monitor for nodejs using a web console based with expressjs

API

inside expressjs project

var app = require('express')();
var monitode = require('monitode');

app.use(monitode({password:'psw'}));

inside nodejs project

var monitode = require('monitode');

monitode({password:'psw'});

monitode(options)

Options

  • port - Port for accept connection (default 30000)
  • password - Password for authentication (default 'admin', 'password')
